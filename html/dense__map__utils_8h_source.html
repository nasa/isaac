98 // A class to store timestamped poses, implementing O(log(n)) linear
99 // interpolation at a desired timestamp. For fast access, keep the
100 // poses in bins obtained by flooring the timestamp, which is measured
101 // in seconds. It is assumed that there are a handful of poses
102 // measured every second, so in each bin. When bins get large, or too
103 // many bins are empty, the efficiency of this algorithm goes down.
104 class StampedPoseStorage {
105  public:
106  void addPose(Eigen::Affine3d const& pose, double timestamp);
107 
108  // Find the interpolated pose by looking up the two poses with
109  // closest timestamps that are below and above input_timestamp. If
110  // the gap between those timestamps is more than max_gap, return
111  // failure, as then likely the interpolation result is not accurate.
112  bool interpPose(double input_timestamp, double max_gap, Eigen::Affine3d& out_pose) const;
113 
114  void clear();
115 
116  bool empty() const;
117 
118  private:
119  std::map<int, std::map<double, Eigen::Affine3d> > m_poses;
120 };
